import { Kbd } from "starlight-kbd/components";

| 模式 | 按键                                | 描述                                       |
| ---- | ----------------------------------- | ------------------------------------------ |
| `N`  | <Kbd linux="/" /> `{pattern}`       | 向后搜索输入的 pattern                     |
| `N`  | <Kbd linux="?"/> `{pattern}`        | 向前搜索输入的 pattern                     |
| `N`  | <Kbd linux="n"/> / <Kbd linux="N"/> | 根据搜索的方式向前或向后搜索输入的 pattern |
| `N`  | <Kbd linux="*"/>                    | 快速向后搜索当前光标下的单词               |
| `N`  | <Kbd linux="#"/>                    | 快速向前搜索当前光标下的单词               |
