import { Kbd } from "starlight-kbd/components";

| 模式 | 按键                                     | 描述                                             |
| ---- | ---------------------------------------- | ------------------------------------------------ |
| `N`  | <Kbd linux="ctrl+w" /> <Kbd linux="s" /> | 水平分屏                                         |
| `N`  | <Kbd linux="ctrl+w" /> <Kbd linux="v" /> | 垂直分屏                                         |
| `N`  | <Kbd linux="ctrl+w" /> <Kbd linux="+" /> | 增加分屏高度                                     |
| `N`  | <Kbd linux="ctrl+w" /> <Kbd linux="-" /> | 减少分屏高度                                     |
| `N`  | <Kbd linux="ctrl+w" /> <Kbd linux=">" /> | 减少分屏宽度                                     |
| `N`  | <Kbd linux="ctrl+w" /> <Kbd linux="<" /> | 增加分屏宽度                                     |
| `N`  | <Kbd linux="ctrl+w" /> <Kbd linux="=" /> | 将所有分屏设置为相同的宽度和高度                 |
| `N`  | <Kbd linux="ctrl+w" /> <Kbd linux="c" /> | 关闭当前分屏                                     |
| `N`  | <Kbd linux="ctrl+w" /> <Kbd linux="r" /> | 旋转所有分屏                                     |
| `C`  | `:sp`                                    | 和 <Kbd linux="ctrl+w" /> <Kbd linux="s" /> 相同 |
| `C`  | `:vs`                                    | 和 <Kbd linux="ctrl+w" /> <Kbd linux="v" /> 相同 |
